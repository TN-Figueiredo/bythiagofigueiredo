import type { CSSProperties } from 'react'
import type { AuthTheme } from './types'
import { ADMIN_THEME_DEFAULT } from './types'

export function mergeTheme(override: Partial<AuthTheme> | undefined): AuthTheme {
  if (!override) return ADMIN_THEME_DEFAULT
  return { ...ADMIN_THEME_DEFAULT, ...override }
}

export function buildThemeVars(theme: AuthTheme): CSSProperties {
  return {
    '--auth-bg': theme.bg,
    '--auth-card-bg': theme.card,
    '--auth-accent': theme.accent,
    '--auth-accent-hover': theme.accentHover,
    '--auth-text': theme.text,
    '--auth-muted': theme.muted,
    '--auth-border': theme.border,
  } as CSSProperties
}

export function AdminLogin({
  actions,
  strings: stringsOverride,
  locale,
  logo,
  footer,
  theme: themeOverride,
  turnstile: turnstileConfig,
  emailHint,
  authError,
}: AuthPageProps) {
  const router = useRouter()
  const searchParams = useSearchParams()

  const baseStrings = getAdminAuthStrings(locale)
  const strings: AuthStrings = stringsOverride
    ? { ...baseStrings, ...stringsOverride }
    : baseStrings

  const theme = mergeTheme(themeOverride)
  const cssVars = buildThemeVars(theme)

  const [email, setEmail] = useState(emailHint ?? '')
  const [password, setPassword] = useState('')
  const [showPassword, setShowPassword] = useState(false)
  const [loading, setLoading] = useState(false)
  const [googleLoading, setGoogleLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [turnstileToken, setTurnstileToken] = useState<string | null>(null)

  const emailRef = useRef<HTMLInputElement>(null)
  const passwordRef = useRef<HTMLInputElement>(null)
  const turnstileRef = useRef<HTMLDivElement>(null)
  const widgetIdRef = useRef<string | null>(null)
  const hasError = !!error || !!authError

  // ── Compute displayed error ──────────────────────────────────────────────
  const displayError = error
    ?? (authError === 'auth' ? strings.errorGeneric : undefined)
    ?? (authError === 'unauthorized' ? strings.errorGeneric : undefined)
    ?? (authError ? strings.errorGeneric : undefined)

  // ── Mount Turnstile ──────────────────────────────────────────────────────
  useEffect(() => {
    if (!turnstileConfig || !turnstileRef.current) return
    const script = document.createElement('script')
    script.src = 'https://challenges.cloudflare.com/turnstile/v0/api.js'
    script.async = true
    script.defer = true
    script.onload = () => {
      if (window.turnstile && turnstileRef.current) {
        const id = window.turnstile.render(turnstileRef.current, {
          sitekey: turnstileConfig.siteKey,
          callback: (t: string) => setTurnstileToken(t),
        })
        widgetIdRef.current = id
      }
    }
    document.head.appendChild(script)
    return () => { script.remove() }
  }, [turnstileConfig])

  const resetTurnstile = useCallback(() => {
    if (window.turnstile && widgetIdRef.current) {
      window.turnstile.reset(widgetIdRef.current)
    }
    setTurnstileToken(null)
  }, [])

  // ── Handlers ─────────────────────────────────────────────────────────────
  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setError(null)

    if (turnstileConfig && !turnstileToken) {
      setError(strings.errorTurnstileLoading)
      return
    }

    setLoading(true)
    try {
      const result = await actions.signInWithPassword({
        email,
        password,
        turnstileToken: turnstileToken ?? undefined,
      })

      if (!result.ok) {
        const msg = result.error === 'invalid_credentials'
          ? strings.errorInvalidCredentials
          : strings.errorGeneric
        setError(msg)
        resetTurnstile()
        // Focus management: move to first invalid field on error
        passwordRef.current?.focus()
        return
      }

      const next = searchParams.get('next')
      if (next) {
        router.push(next)
      } else {
        router.push('/admin')
      }
      router.refresh()
    } finally {
      setLoading(false)
    }
  }

  async function handleGoogleSignIn() {
    setGoogleLoading(true)
    setError(null)
    try {
      await actions.signInWithGoogle({ redirectTo: searchParams.get('next') ?? undefined })
    } finally {
      setGoogleLoading(false)
    }
  }

  // ── Render ────────────────────────────────────────────────────────────────
  return (
    <div
      className="min-h-screen flex items-center justify-center bg-[var(--auth-bg)]"
      style={cssVars as CSSProperties}
    >
      <div className="max-w-md w-full px-4">
        {logo && <div className="flex justify-center mb-6">{logo}</div>}

        <div className="bg-[var(--auth-card-bg)] rounded-xl p-8 shadow-sm border border-[var(--auth-border)]">
          <div className="text-center mb-6">
            <h1 className="text-2xl font-bold text-[var(--auth-text)]">{strings.title}</h1>
            <p className="text-sm text-[var(--auth-muted)] mt-1">{strings.subtitle}</p>
          </div>

          {/* Error region */}
          {displayError && (
            <div
              role="alert"
              aria-live="assertive"
              className="bg-red-50 text-red-700 border border-red-200 p-3 rounded-lg mb-4 text-sm"
            >
              {displayError}
            </div>
          )}

          {/* Google button */}
          <button
            type="button"
            onClick={handleGoogleSignIn}
            disabled={googleLoading || loading}
            className="w-full flex items-center justify-center gap-3 border border-[var(--auth-border)] hover:bg-slate-50 disabled:opacity-60 py-3 rounded-lg font-medium transition text-[var(--auth-text)] mb-6"
          >
            <GoogleIcon />
            {googleLoading ? strings.googleButtonLoading : strings.googleButton}
          </button>

          {/* Divider */}
          <div className="relative mb-6" aria-hidden="true">
            <div className="absolute inset-0 flex items-center">
              <div className="w-full border-t border-[var(--auth-border)]" />

            <div className="relative flex justify-center text-sm">
              <span className="px-4 bg-[var(--auth-card-bg)] text-[var(--auth-muted)]">
                {strings.divider}
              </span>
            </div>
          </div>

          {/* Email+password form */}
          <form onSubmit={handleSubmit} noValidate>
            <div className="space-y-4">
              {/* Email */}
              <div>

                  value={email}

                  autoComplete="email"
                  required
                  placeholder={strings.emailPlaceholder}
                  className="w-full border border-[var(--auth-border)] rounded-lg px-4 py-2.5 text-sm text-[var(--auth-text)] outline-none focus:ring-2 focus:ring-[var(--auth-accent)]"
                />

import { mergeTheme, buildThemeVars } from './theme-utils'

function validatePassword(pw: string): string | null {
  if (pw.length < 8) return 'weak'
  if (!/[A-Za-z]/.test(pw)) return 'weak'
  if (!/\d/.test(pw)) return 'weak'
  return null
}

export function AdminResetPassword({
  actions,
  strings: stringsOverride,
  locale,
  logo,
  footer,
  theme: themeOverride,
  redirectAfterReset = '/admin',
}: ResetPasswordPageProps) {

  const baseStrings = getAdminResetPasswordStrings(locale)

    ? { ...baseStrings, ...stringsOverride }
    : baseStrings

  const cssVars = buildThemeVars(theme)

  const [canReset, setCanReset] = useState(false)
  const [password, setPassword] = useState('')
  const [confirm, setConfirm] = useState('')
  const [error, setError] = useState<string | null>(null)
  const [loading, setLoading] = useState(false)
  const [success, setSuccess] = useState(false)

  useEffect(() => {
    if (!actions.onAuthStateChange) return

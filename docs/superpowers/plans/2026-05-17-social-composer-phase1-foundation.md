# Phase 1: Foundation & Core Backend

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create the database schema, caption variable resolution, pipeline redesign, and Bluesky provider fixes.

**Architecture:** Extends the existing Social Hub schema (3 tables from `20260513100000`, extended by `20260514100000` and `20260515100000`) with 3 new tables (`social_templates`, `post_metrics`, `link_in_bio_entries`) and 5 ALTER TABLE operations. Caption variable system is a pure function (no DB dependency) that resolves `{{link}}`, `{{title}}`, `{{url}}` placeholders at publish time. Pipeline step `og_scrape` is renamed to `platform_prepare` throughout the codebase and DB (RPC validation function, JSONB pipeline_steps data). Bluesky auth switches from app-password-per-request to stored JWT with lazy refresh.

**Tech Stack:** Next.js 15, TypeScript 5 strict, Supabase (PostgreSQL 17), Vitest, `@atproto/api` (Bluesky SDK)

**Spec:** `docs/superpowers/specs/2026-05-17-social-composer-stories-templates-design.md` (Sections 2, 3, 6.3, 8)

**Migration numbering:** Continues from `20260517000005`. Use `npm run db:new social_composer_v6` to generate the next sequential timestamp.

CREATE OR REPLACE FUNCTION public.update_pipeline_step(
  p_post_id   UUID,
  p_step_name TEXT,
  p_patch     JSONB
) RETURNS VOID AS $$
DECLARE
  idx INT;
BEGIN
  -- 1. Validate step name against the known pipeline stages.
  IF p_step_name NOT IN ('post_created', 'short_link', 'platform_prepare', 'deliver') THEN
    RAISE EXCEPTION 'update_pipeline_step: invalid step name "%"', p_step_name
      USING ERRCODE = 'invalid_parameter_value';
  END IF;

  -- 2. Validate that the patch object contains the required fields.
  IF p_patch->>'step' IS NULL OR p_patch->>'status' IS NULL OR p_patch->>'at' IS NULL THEN
    RAISE EXCEPTION 'update_pipeline_step: p_patch must contain "step", "status", and "at" fields'
      USING ERRCODE = 'invalid_parameter_value';
  END IF;

  -- 3. Perform the atomic upsert into pipeline_steps (no read-modify-write race).
  SELECT ordinality - 1 INTO idx
  FROM social_posts,
       jsonb_array_elements(pipeline_steps) WITH ORDINALITY AS e(elem, ordinality)
  WHERE id = p_post_id AND elem->>'step' = p_step_name;

  IF idx IS NOT NULL THEN
    UPDATE social_posts
    SET pipeline_steps = jsonb_set(pipeline_steps, ARRAY[idx::TEXT], p_patch)
    WHERE id = p_post_id;
  ELSE
    UPDATE social_posts
    SET pipeline_steps = pipeline_steps || jsonb_build_array(p_patch)
    WHERE id = p_post_id;
  END IF;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION public.prune_link_in_bio()
RETURNS trigger
LANGUAGE plpgsql
SET search_path = 'public'
AS $$
DECLARE
  entry_count INT;
BEGIN
  SELECT count(*) INTO entry_count
  FROM link_in_bio_entries
  WHERE site_id = NEW.site_id;

  -- If over limit after this insert, delete the oldest
  IF entry_count > 20 THEN

    WHERE id IN (
      SELECT id FROM link_in_bio_entries
      WHERE site_id = NEW.site_id
      ORDER BY position DESC, created_at ASC
      LIMIT (entry_count - 20)
    );
  END IF;

  RETURN NEW;
END;
$$;

// apps/web/src/lib/social/caption-variables.ts
//
// Caption variable system for Social Composer v6.
// Resolves {{link}}, {{title}}, {{url}} placeholders at publish time.
// Unknown variables (e.g., {{foo}}) are left as literal text.
// See spec: Section 2 — Caption Variable System.

import type { Provider } from '@tn-figueiredo/social'

// ---------------------------------------------------------------------------
// Types
// ---------------------------------------------------------------------------

export interface CaptionContext {
  link: string   // short URL like "go.btf.com/abc123"
  title: string  // post title
  url: string    // raw destination URL
}

// ---------------------------------------------------------------------------
// Constants
// ---------------------------------------------------------------------------

/** Placeholder length used when the short link is not yet generated. */
const LINK_PLACEHOLDER_LENGTH = 24

/** Known variable names that will be resolved. */
const KNOWN_VARS = new Set(['link', 'title', 'url'])

/** Regex to match {{variable}} patterns. Only known vars are replaced. */
const VAR_REGEX = /\{\{(link|title|url)\}\}/g

/**
 * Per-platform default caption templates.
 * Pre-filled when a platform is toggled on and its caption field is empty.
 * Instagram uses "Link na bio" because IG captions don't support clickable links.
 */
export const PLATFORM_CAPTION_DEFAULTS: Record<Provider, string> = {
  facebook: '{{title}}\n\n{{link}}',
  bluesky: '{{title}}\n\n{{link}}',
  instagram: '{{title}}\n\nLink na bio',
  youtube: '{{title}}\n\n{{link}}',
}

// ---------------------------------------------------------------------------
// Core functions
// ---------------------------------------------------------------------------

/**
 * Resolve a caption template by substituting known variables with context values.
 * Unknown variables like {{foo}} are left as-is (literal text).
 *
 * @param template  - Raw caption template string from the DB
 * @param context   - Values to substitute into the template
 * @returns Resolved caption string
 */
export function resolveCaption(template: string, context: CaptionContext): string {
  return template.replace(VAR_REGEX, (_match, varName: string) => {
    if (!KNOWN_VARS.has(varName)) return _match
    return context[varName as keyof CaptionContext]
  })
}

/**
 * Calculate the character count of a resolved caption.
 * When the link is not yet known (empty string), uses 24 chars as a placeholder
 * length to match the typical short URL length (e.g., "go.btf.com/abc123").
 *
 * @param template  - Raw caption template string
 * @param context   - Values for substitution (link may be empty)
 * @returns Character count of the resolved string
 */
export function resolvedLength(template: string, context: CaptionContext): number {
  const resolvedContext: CaptionContext = {
    ...context,
    link: context.link || 'x'.repeat(LINK_PLACEHOLDER_LENGTH),
  }
  return resolveCaption(template, resolvedContext).length
}

/**
 * Check if a caption template is missing the {{link}} variable.
 * Returns true if no {{link}} is found — used for a soft warning
 * ("No link in caption") that never blocks publish.
 *
 * @param template - Raw caption template string
 * @returns true if template does NOT contain {{link}}
 */
export function findMissingLink(template: string): boolean {
  return !template.includes('{{link}}')
}

import { BskyAgent } from '@atproto/api'

const DEFAULT_PDS = 'https://bsky.social'

/** Threshold in ms — refresh if token expires within this window. */
const REFRESH_THRESHOLD_MS = 5 * 60 * 1000

export interface BlueskySession {
  did: string
  handle: string
  accessJwt: string
  refreshJwt: string
}

export async function createSession(
  handle: string,
  appPassword: string,
  pdsUrl?: string,
): Promise<BskyAgent> {
  const agent = new BskyAgent({ service: pdsUrl ?? DEFAULT_PDS })
  await agent.login({ identifier: handle, password: appPassword })
  return agent
}

export async function resumeSession(
  agent: BskyAgent,
  session: BlueskySession,
): Promise<BskyAgent> {
  await agent.resumeSession({
    did: session.did,
    handle: session.handle,
    accessJwt: session.accessJwt,
    refreshJwt: session.refreshJwt,
    active: true,
  })
  return agent
}

/**
 * Check if a JWT is expiring within the threshold window (5 minutes).
 * Returns true if the token should be refreshed before making API calls.
 */
export function isJwtExpiringSoon(expiresAt: string | null | undefined): boolean {
  if (!expiresAt) return true
  const expiresMs = new Date(expiresAt).getTime()
  return expiresMs - Date.now() < REFRESH_THRESHOLD_MS
}

/**
 * Refresh a Bluesky session using the refresh JWT.
 * Calls `com.atproto.server.refreshSession` via the AT Protocol API.
 *
 * @param currentSession - The current session with an active refresh JWT
 * @param pdsUrl         - Optional PDS URL (defaults to bsky.social)
 * @returns Updated session with new access and refresh JWTs
 */
export async function refreshSession(
  currentSession: BlueskySession,
  pdsUrl?: string,
): Promise<BlueskySession> {
  const agent = new BskyAgent({ service: pdsUrl ?? DEFAULT_PDS })

  // Resume with the current tokens so the agent has the refresh JWT
  await agent.resumeSession({
    did: currentSession.did,
    handle: currentSession.handle,
    accessJwt: currentSession.accessJwt,
    refreshJwt: currentSession.refreshJwt,
    active: true,
  })

  // Call the AT Protocol refresh endpoint
  const response = await agent.api.com.atproto.server.refreshSession(undefined, {
    headers: {
      Authorization: `Bearer ${currentSession.refreshJwt}`,
    },
  })

  return {
    did: response.data.did,
    handle: response.data.handle,
    accessJwt: response.data.accessJwt,
    refreshJwt: response.data.refreshJwt,
  }
}

  supabase: ReturnType<typeof getSupabaseServiceClient>,

    id: unknown
    access_token_enc: unknown
    refresh_token_enc: unknown
    token_expires_at: unknown

  key: Buffer,

  oldExpiresAt: string,
): Promise<FreshToken> {

  // The bluesky_refresh_jwt_enc column stores the AT Protocol refresh token.
  const connRecord = conn as unknown as Record<string, unknown>
  const refreshJwt = connRecord.bluesky_refresh_jwt_enc as string | null
  const accessJwt = connRecord.bluesky_access_jwt_enc as string | null
  const did = connRecord.bluesky_did as string | null

    // App passwords don't expire so this should not normally reach here.
    const accessToken = decrypt(conn.access_token_enc as string, key)
    return { accessToken, connectionId }
  }

    const { refreshSession } = await import(
      '@tn-figueiredo/social/providers/bluesky'

    }

    // Return the app password as the "access token" for the provider.
    // The provider still uses createSession(handle, appPassword) for API calls.
    const accessToken = decrypt(conn.access_token_enc as string, key)
    return { accessToken, connectionId }
  } catch (err) {
    const error = err instanceof Error ? err : new Error(String(err))

    if (
      error.message.includes('InvalidToken') ||
      error.message.includes('ExpiredToken')
    ) {
      await markConnectionRevoked(connectionId)
      throw new TokenRevokedError('bluesky', connectionId)
    }

    throw new Error(`Bluesky token refresh failed: ${error.message}`)
  }
}
